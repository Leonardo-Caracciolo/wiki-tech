Creacionales

Singleton Creacional

Cuándo: Una sola instancia en toda la app — configuración, conexión a DB, logger.

class DatabaseConnection:
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance.connection = create_connection()
        return cls._instance

# Ambas variables apuntan a la misma instancia
db1 = DatabaseConnection()
db2 = DatabaseConnection()
assert db1 is db2  # True
Factory Method Creacional

Cuándo: Crear objetos sin especificar la clase exacta. El equipo lo usa para el parser de documentos.

from abc import ABC, abstractmethod

class DocumentParser(ABC):
    @abstractmethod
    def parse(self, filepath: str) -> str: ...

class MarkdownParser(DocumentParser):
    def parse(self, filepath): return parse_md(filepath)

class DocxParser(DocumentParser):
    def parse(self, filepath): return parse_docx(filepath)

def get_parser(extension: str) -> DocumentParser:
    parsers = {"md": MarkdownParser, "docx": DocxParser}
    return parsers[extension]()
Builder Creacional

Cuándo: Construir objetos complejos paso a paso. Reportes, queries dinámicas, emails.

class ReportBuilder:
    def __init__(self):
        self._report = {}

    def with_title(self, title: str) -> "ReportBuilder":
        self._report["title"] = title
        return self

    def with_data(self, data) -> "ReportBuilder":
        self._report["data"] = data
        return self

    def build(self) -> dict:
        return self._report

# Uso: method chaining
report = ReportBuilder().with_title("Ratio Q1").with_data(df).build()

Estructurales

Adapter Estructural

Cuándo: Integrar una interfaz incompatible. Lo usamos al migrar de pywin32 a python-docx.

# Interfaz vieja (pywin32)
class LegacyWordCOM:
    def open_document(self, path): ...
    def save_as_pdf(self, path): ...

# Interfaz nueva que queremos usar
class WordAdapter:
    def __init__(self, legacy: LegacyWordCOM):
        self._legacy = legacy

    def render(self, path) -> bytes:
        self._legacy.open_document(path)
        return self._legacy.save_as_pdf(path)
Decorator Estructural

Cuándo: Agregar comportamiento sin modificar la clase original. Logging, caché, validación.

import functools, logging

def log_execution(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        logging.info(f"Ejecutando {func.__name__}")
        result = func(*args, **kwargs)
        logging.info(f"{func.__name__} finalizado")
        return result
    return wrapper

@log_execution
def generar_ratio_report(cliente_id: int): ...
Facade Estructural

Cuándo: Simplificar una interfaz compleja. Un punto de entrada que orquesta múltiples subsistemas.

class RatioReportFacade:
    """Un método público que orquesta todo el proceso."""
    def __init__(self):
        self._reader   = ExcelReader()
        self._parser   = DataParser()
        self._renderer = DocxRenderer()

    def generate(self, excel_path: str, output_path: str):
        raw   = self._reader.read(excel_path)
        data  = self._parser.parse(raw)
        self._renderer.render(data, output_path)

Comportamiento

Strategy Comportamiento

Cuándo: Intercambiar algoritmos en runtime. Diferentes formatos de export, diferentes parsers por tipo PLI.

from abc import ABC, abstractmethod

class ExportStrategy(ABC):
    @abstractmethod
    def export(self, data) -> bytes: ...

class ExcelExport(ExportStrategy):
    def export(self, data) -> bytes: return to_excel(data)

class PDFExport(ExportStrategy):
    def export(self, data) -> bytes: return to_pdf(data)

class ReportExporter:
    def __init__(self, strategy: ExportStrategy):
        self._strategy = strategy

    def export(self, data): return self._strategy.export(data)
Observer Comportamiento

Cuándo: Notificar cambios a múltiples oyentes. Eventos de bot, notificaciones de proceso, webhooks.

class EventBus:
    def __init__(self):
        self._listeners = {}

    def subscribe(self, event: str, handler):
        self._listeners.setdefault(event, []).append(handler)

    def publish(self, event: str, data=None):
        for handler in self._listeners.get(event, []):
            handler(data)

bus = EventBus()
bus.subscribe("bot_completed", send_email_notification)
bus.subscribe("bot_completed", log_to_database)
bus.publish("bot_completed", {"bot": "ratio-report", "status": "ok"})
💡 Regla práctica: Si tenés un if/elif largo para elegir comportamiento → Strategy. Si necesitás notificar a múltiples partes → Observer. Si la API es compleja → Facade.
