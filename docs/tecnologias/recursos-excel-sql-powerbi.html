Normalización de datos

Código de referencia — merge y limpieza

import pandas as pd

# Leer Excel con múltiples hojas
df_clientes  = pd.read_excel("datos.xlsx", sheet_name="Clientes", header=0)
df_ventas    = pd.read_excel("datos.xlsx", sheet_name="Ventas",   dtype={"CUIT": str})

# Limpieza estándar
df_clientes["nombre"] = df_clientes["nombre"].str.strip().str.replace(r"\s+", " ", regex=True)
df_ventas["monto"]    = pd.to_numeric(df_ventas["monto"], errors="coerce")
df_ventas               = df_ventas.dropna(subset=["id_cliente", "monto"])

# Merge — todos los clientes, hayan vendido o no
df = pd.merge(df_clientes, df_ventas, on="id_cliente", how="left")

# Agrupar para Power BI
resumen = df.groupby(["region", "producto"]).agg(
    total_ventas=("monto", "sum"),
    cant_facturas=("id_factura", "count"),
).reset_index()

from sqlalchemy import create_engine
import pandas as pd

# String de conexión SQL Server con SQLAlchemy
engine = create_engine(
    "mssql+pyodbc://usuario:contraseña@servidor/base_datos"
    "?driver=ODBC+Driver+17+for+SQL+Server"
)

# Cargar DataFrame a SQL Server
df.to_sql(
    name="ventas",
    con=engine,
    if_exists="append",   # append = no borrar datos existentes
    index=False,
    chunksize=1000,       # batch para mejor performance
)

Pipeline idempotente — estructura recomendada

import logging
from pydantic import BaseModel, ValidationError

logger = logging.getLogger(__name__)

class RowSchema(BaseModel):
    id_cliente: int
    cuit:       str
    monto:      float
    fecha:      str

def run_pipeline(excel_path: str, engine) -> dict:
    """Pipeline idempotente: leer → validar → limpiar → cargar."""
    stats = {"ok": 0, "skipped": 0, "errors": []}

    df_raw = pd.read_excel(excel_path)
    rows_validas = []

    for _, row in df_raw.iterrows():
        try:
            validated = RowSchema(**row.to_dict())
            rows_validas.append(validated.model_dump())
            stats["ok"] += 1
        except ValidationError as e:
            logger.warning(f"Fila inválida: {e}")
            stats["errors"].append(str(e))
            stats["skipped"] += 1

    if rows_validas:
        pd.DataFrame(rows_validas).to_sql(
            "ventas", engine, if_exists="append", index=False
        )

    logger.info(f"Pipeline completo — OK: {stats['ok']} | Saltados: {stats['skipped']}")
    return stats
