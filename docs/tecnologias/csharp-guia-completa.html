/// <summary>
/// Procesa y valida un comprobante fiscal contra los datos de ARCA.
/// </summary>
/// <param name="comprobante">El comprobante a validar. No puede ser null.</param>
/// <param name="cuitReceptor">CUIT del receptor sin guiones (11 dígitos).</param>
/// <returns>
/// Un <see cref="ValidationResult"/> con IsValid=true si el comprobante es válido.
/// </returns>
/// <exception cref="ArgumentNullException">Si comprobante es null.</exception>
/// <exception cref="FormatException">Si cuitReceptor no tiene 11 dígitos.</exception>
/// <example>
/// <code>
/// var result = await validador.ValidarAsync(comprobante, "20123456789");
/// if (result.IsValid) Console.WriteLine("OK");
/// </code>
/// </example>
public async Task<ValidationResult> ValidarAsync(
    Comprobante comprobante,
    string cuitReceptor,
    CancellationToken ct = default)
{
    ArgumentNullException.ThrowIfNull(comprobante);
    if (cuitReceptor?.Length != 11)
        throw new FormatException("CUIT debe tener exactamente 11 dígitos");

    return await _arcaClient.ConsultarAsync(comprobante, cuitReceptor, ct);
}

Async / Await — Buenas prácticas

// ✅ Siempre propagar CancellationToken
public async Task<List<Cliente>> GetClientesAsync(
    CancellationToken ct = default)
{
    return await _db.Clientes
        .Where(c => c.Activo)
        .ToListAsync(ct);  // ct en cada operación async
}

// ✅ ConfigureAwait(false) en librerías
var datos = await _service.GetAsync().ConfigureAwait(false);

// ✅ Ejecutar en paralelo con Task.WhenAll
var (clientes, productos) = await Task.WhenAll(
    _clienteService.GetAllAsync(ct),
    _productoService.GetAllAsync(ct)
);

LINQ — Patrones más usados

var clientes = new List<Cliente> { /* ... */ };

// Filtrar, proyectar y ordenar
var nombres = clientes
    .Where(c => c.Activo && c.Monto > 10_000)
    .OrderByDescending(c => c.Monto)
    .Select(c => new { c.Id, c.Nombre, c.Monto })
    .Take(10)
    .ToList();

// GroupBy
var porRegion = clientes
    .GroupBy(c => c.Region)
    .Select(g => new {
        Region = g.Key,
        Total  = g.Sum(c => c.Monto),
        Cant   = g.Count()
    });

// Chequear existencia
bool tieneVip = clientes.Any(c => c.Monto > 100_000);
bool todosActivos = clientes.All(c => c.Activo);

// Aplanar colecciones anidadas
var todosOrders = clientes
    .SelectMany(c => c.Ordenes)
    .Where(o => o.Estado == "Pendiente")
    .ToList();

// Switch expression con patterns
string GetEstadoLabel(Pedido pedido) => pedido.Estado switch
{
    Estado.Pendiente   => "⏳ Pendiente",
    Estado.EnProceso   => "🔄 En proceso",
    Estado.Completado  => "✅ Completado",
    Estado.Cancelado   => "❌ Cancelado",
    _                  => "❓ Desconocido"
};

// Property pattern
bool EsClienteVip(Cliente c) => c is
    { Activo: true, Monto: > 100_000, Region: "ARG" };

// Deconstruct
