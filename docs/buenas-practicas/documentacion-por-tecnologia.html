Docstring Google-style completo — ejemplo real del equipo (módulo ARCA)

"""
Módulo para procesamiento y validación de comprobantes ARCA.

Este módulo provee utilidades para descargar, parsear y validar
comprobantes electrónicos obtenidos mediante los web services de ARCA.
"""
from dataclasses import dataclass
from datetime import date
from typing import Optional

@dataclass
class Invoice:
    """Representa un comprobante electrónico de ARCA.

    Attributes:
        cuit_emisor: CUIT del emisor sin guiones (11 dígitos).
        numero: Número de comprobante (punto de venta + número).
        fecha: Fecha de emisión del comprobante.
        importe_total: Importe total incluyendo impuestos.
        cae: Código de Autorización Electrónico otorgado por ARCA.
        tipo: Tipo de comprobante (A, B, C, M, E).
    """
    cuit_emisor:   str
    numero:        str
    fecha:         date
    importe_total: float
    cae:           str
    tipo:          str


def validar_comprobante(invoice: Invoice, cuit_receptor: str) -> bool:
    """Valida un comprobante electrónico contra los criterios del receptor.

    Verifica que el comprobante sea del tipo correcto según la condición
    fiscal del receptor y que el CAE sea formalmente válido (no verifica
    contra ARCA, solo formato local).

    Args:
        invoice: Instancia del comprobante a validar.
        cuit_receptor: CUIT del receptor sin guiones (11 dígitos).

    Returns:
        True si el comprobante pasa todas las validaciones, False en caso
        contrario.

    Raises:
        ValueError: Si el CUIT del receptor no tiene exactamente 11 dígitos.
        TypeError: Si invoice no es una instancia de Invoice.

    Example:
        >>> from datetime import date
        >>> inv = Invoice(
        ...     cuit_emisor="20123456789",
        ...     numero="0001-00001234",
        ...     fecha=date(2024, 3, 1),
        ...     importe_total=12100.00,
        ...     cae="74123456789012",
        ...     tipo="A"
        ... )
        >>> validar_comprobante(inv, "30987654321")
        True
    """
    if not isinstance(invoice, Invoice):
        raise TypeError(f"Se esperaba Invoice, se recibió {type(invoice).__name__}")
    if len(cuit_receptor.replace("-", "")) != 11:
        raise ValueError(f"CUIT inválido: {cuit_receptor!r}. Debe tener 11 dígitos.")
    if len(invoice.cae) != 14:
        return False
    if invoice.importe_total <= 0:
        return False
    return True

Opción A — Rubberduck VBA (recomendado)

' @ModuleDescription("Procesamiento de comprobantes para reportes de Transfer Pricing.")
' @Folder("Comprobantes")
Option Explicit

' @Description("Tipo que representa un comprobante fiscal.")
Public Type TComprobante
    CUIT        As String
    Numero      As String
    Fecha       As Date
    ImporteTotal As Double
    TipoComp    As String
End Type

' @Description("Valida que un comprobante tenga todos los campos requeridos.")
' @Param comprobante  "Estructura TComprobante a validar."
' @Param cuitReceptor "CUIT del receptor sin guiones (11 dígitos)."
' @Returns "True si es válido, False en caso contrario."
' @Remarks "No verifica el CAE contra ARCA, solo validación de formato local."
Public Function ValidarComprobante(ByVal comprobante As TComprobante, _
                                     ByVal cuitReceptor As String) As Boolean
    ValidarComprobante = False
    If Len(Replace(cuitReceptor, "-", "")) <> 11 Then Exit Function
    If Len(Trim(comprobante.CUIT)) = 0 Then Exit Function
    If comprobante.ImporteTotal <= 0 Then Exit Function
    ValidarComprobante = True
End Function

⚡ C# — XML Docs + DocFX

C# tiene documentación XML nativa integrada al compilador. Visual Studio ya la muestra como IntelliSense al hacer hover. Con DocFX se convierte en un sitio web estático.

XML Docs — tags esenciales

TagPropósito
<summary>Descripción corta — aparece en IntelliSense al hacer hover
<param name="x">Descripción del parámetro x
<returns>Descripción del valor de retorno
<exception cref="Ex">Excepción que puede lanzar y cuándo
<example><code>Ejemplo de uso del método
<see cref="Clase"/>Referencia cruzada a otra clase o método
<remarks>Información adicional extendida

XML Docs completo — ejemplo real del equipo

/// <summary>
/// Procesa y valida un comprobante fiscal contra los datos de ARCA.
/// </summary>
/// <param name="comprobante">El comprobante a validar. No puede ser null.</param>
/// <param name="cuitReceptor">CUIT del receptor sin guiones (11 dígitos).</param>
/// <param name="ct">Token de cancelación para operaciones asíncronas.</param>
/// <returns>
/// Un <see cref="ValidationResult"/> con IsValid=true si el comprobante
/// pasa todas las validaciones locales y de ARCA.
/// </returns>
/// <exception cref="ArgumentNullException">Si comprobante es null.</exception>
/// <exception cref="FormatException">Si cuitReceptor no tiene exactamente 11 dígitos.</exception>
/// <example>
/// <code>
/// var result = await validador.ValidarAsync(comprobante, "20123456789");
/// if (result.IsValid) Console.WriteLine("Comprobante válido");
/// </code>
/// </example>
public async Task<ValidationResult> ValidarAsync(
    Comprobante comprobante,
    string cuitReceptor,
    CancellationToken ct = default)
{
    ArgumentNullException.ThrowIfNull(comprobante);
    if (cuitReceptor?.Length != 11)
        throw new FormatException("CUIT debe tener exactamente 11 dígitos");
    return await _arcaClient.ConsultarAsync(comprobante, cuitReceptor, ct);
}

Generar el sitio con DocFX

# Instalar DocFX
dotnet tool install -g docfx

# Inicializar en el proyecto
docfx init -q

# docfx.json — configuración mínima
{
  "metadata": [{
    "src": [{ "files": ["**/*.csproj"], "src": "../src" }],
    "dest": "api"
  }],
  "build": {
    "content": [
      { "files": ["api/**.yml", "api/index.md"] },
      { "files": ["docs/**.md", "*.md"] }
    ],
    "dest": "_site",
    "globalMetadata": {
      "_appName": "Wiki BPS",
      "_enableSearch": true
    }
  }
}

# Generar y previsualizar
docfx docfx.json --serve   # → http://localhost:8080

# Solo generar (para CI/CD)
docfx docfx.json
✅ Activar <GenerateDocumentationFile>true</GenerateDocumentationFile> en el .csproj para que el compilador genere el XML de documentación automáticamente en el build.
