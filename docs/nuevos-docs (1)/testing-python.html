pytest — Template base del equipo

"""
Template base para tests unitarios con pytest.

Convenciones BPS:
  - Un archivo test_módulo.py por módulo a testear
  - Una clase TestXxx por clase/función principal
  - Método: test_<caso>_<condición>_<resultado_esperado>
  - Fixtures compartidas en conftest.py
  - Mocks para TODAS las dependencias externas
"""
import pytest
from unittest.mock import MagicMock, patch
from datetime import date

from src.mi_proyecto.processors import validar_comprobante, InvoiceProcessor
from src.mi_proyecto.models import Comprobante


# ── FIXTURES ──────────────────────────────────────────────────────────────────

@pytest.fixture
def comprobante_valido() -> Comprobante:
    """Comprobante válido para happy path tests."""
    return Comprobante(
        cuit_emisor="20123456789",
        numero="0001-00001234",
        fecha=date(2025, 3, 1),
        importe=12100.00,
        tipo="A",
    )

@pytest.fixture
def mock_arca_client():
    """Mock del cliente ARCA para no hacer llamadas reales."""
    with patch("src.mi_proyecto.processors.ArcaClient") as mock:
        mock.return_value.consultar.return_value = {"valido": True}
        yield mock


# ── TESTS ─────────────────────────────────────────────────────────────────────

class TestValidarComprobante:
    """Tests para la función validar_comprobante."""

    def test_comprobante_valido_retorna_true(self, comprobante_valido):
        """Happy path — comprobante con todos los campos correctos."""
        resultado = validar_comprobante(comprobante_valido, "30987654321")
        assert resultado is True

    def test_importe_negativo_retorna_false(self, comprobante_valido):
        comprobante_valido.importe = -100.0
        assert validar_comprobante(comprobante_valido, "30987654321") is False

    def test_cuit_invalido_lanza_value_error(self, comprobante_valido):
        with pytest.raises(ValueError, match="11 dígitos"):
            validar_comprobante(comprobante_valido, "123")

    @pytest.mark.parametrize("importe", [0.0, -1.0, -999.99])
    def test_importes_invalidos_retornan_false(self, comprobante_valido, importe):
        """Parametrize — mismo test para múltiples valores."""
        comprobante_valido.importe = importe
        assert validar_comprobante(comprobante_valido, "30987654321") is False


class TestInvoiceProcessor:
    """Tests con mocking de dependencias externas."""

    def test_procesar_llama_a_arca(self, comprobante_valido, mock_arca_client):
        processor = InvoiceProcessor()
        processor.procesar(comprobante_valido)
        mock_arca_client.return_value.consultar.assert_called_once()

    def test_procesar_retorna_false_si_arca_falla(self, comprobante_valido):
        with patch("src.mi_proyecto.processors.ArcaClient") as mock:
            mock.return_value.consultar.side_effect = ConnectionError("timeout")
            processor = InvoiceProcessor()
            resultado = processor.procesar(comprobante_valido)
            assert resultado is False

conftest.py — Fixtures compartidas

# tests/conftest.py
import pytest
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

@pytest.fixture(scope="session")
def engine():
    """Engine de SQLite en memoria — cero setup, tests rápidos."""
    return create_engine("sqlite:///:memory:")

@pytest.fixture(scope="function")
def db_session(engine):
    """Sesión de DB con rollback automático después de cada test."""
    connection = engine.connect()
    transaction = connection.begin()
    Session = sessionmaker(bind=connection)
    session = Session()
    yield session
    session.close()
    transaction.rollback()  # ← cada test arranca con DB limpia
    connection.close()

Smoke Tests — verificación rápida de producción

Los smoke tests son una suite mínima que verifica que el sistema está vivo y las funciones críticas responden. Se corren después de cada deploy.

# tests/smoke/test_smoke.py
"""
Smoke tests — verificación post-deploy.
Correr con: pytest tests/smoke/ -m smoke

Estos tests NUNCA deben:
  - Modificar datos de producción
  - Ser lentos (máximo 30 segundos total)
  - Fallar por datos variables (ej: fecha del día)
"""
import pytest
import requests

BASE_URL = "http://localhost:8000"

@pytest.mark.smoke
def test_health_endpoint_responde():
    """La API debe estar viva."""
    resp = requests.get(f"{BASE_URL}/health", timeout=5)
    assert resp.status_code == 200

@pytest.mark.smoke
def test_db_connection_ok():
    """La DB debe estar accesible."""
    resp = requests.get(f"{BASE_URL}/health/db", timeout=5)
    assert resp.json()["status"] == "ok"

@pytest.mark.smoke
def test_endpoint_clientes_responde():
    """El endpoint crítico de clientes debe responder."""
    resp = requests.get(f"{BASE_URL}/api/clientes", timeout=10)
    assert resp.status_code in (200, 401)  # OK o requiere auth — ambos son "vivo"

Robot Framework — Tests E2E

Robot Framework es el estándar del equipo para tests de aceptación y E2E. Su sintaxis es legible por no-técnicos.

*** Settings ***
Library    RequestsLibrary
Library    Collections
Library    OperatingSystem
Suite Setup      Crear Sesión API
Suite Teardown   Cerrar Sesión

*** Variables ***
${BASE_URL}       http://localhost:8000
${CUIT_VALIDO}    20123456789

*** Test Cases ***
Flujo Completo Validación de Comprobante
    [Documentation]    Test E2E del flujo de validación
    Dado que tengo un comprobante válido
    Cuando envío el comprobante a validar
    Entonces la respuesta es exitosa
    Y el resultado contiene valido=True

*** Keywords ***
Crear Sesión API
    Create Session    api    ${BASE_URL}

Dado que tengo un comprobante válido
    ${comp}=    Create Dictionary
    ...    cuit=${CUIT_VALIDO}    importe=12100    tipo=A
    Set Suite Variable    ${COMPROBANTE}    ${comp}

Cuando envío el comprobante a validar
    ${resp}=    POST On Session    api    /api/validar    json=${COMPROBANTE}
    Set Suite Variable    ${RESPONSE}    ${resp}

Entonces la respuesta es exitosa
    Should Be Equal As Integers    ${RESPONSE.status_code}    200

Y el resultado contiene valido=True
    ${body}=    Set Variable    ${RESPONSE.json()}
    Should Be True    ${body['valido']}
# Correr Robot Framework
robot tests/e2e/                     # todos los tests
robot --include smoke tests/          # solo smoke
robot --output results/ tests/        # guardar resultados

Testing avanzado — Mocking y Test Doubles

from unittest.mock import MagicMock, patch, call
from pytest_mock import MockerFixture

# ── MagicMock básico ──────────────────────────────────────────
mock_service = MagicMock()
mock_service.get_datos.return_value = [{"id": 1, "nombre": "Test"}]
mock_service.get_datos.assert_called_once_with(42)

# ── patch como decorator ──────────────────────────────────────
@patch("mymodule.requests.get")
def test_api_call(mock_get):
    mock_get.return_value.json.return_value = {"status": "ok"}
    resultado = mi_funcion_que_llama_api()
    assert resultado == "ok"

# ── spy con mocker (pytest-mock) ──────────────────────────────
def test_con_spy(mocker: MockerFixture):
    spy = mocker.spy(mi_objeto, "mi_metodo")
    mi_objeto.mi_metodo("arg")
    spy.assert_called_once_with("arg")

# ── hypothesis — property-based testing ──────────────────────
from hypothesis import given, strategies as st

@given(st.floats(min_value=0, max_value=1_000_000))
def test_calculo_margen_nunca_supera_uno(precio):
    margen = calcular_margen(precio, precio * 0.6)
    assert 0 <= margen <= 1
# Comandos clave de pytest
pytest                              # todos los tests
pytest -v                           # verbose
pytest -k "validar"                 # solo tests con "validar" en el nombre
pytest -m smoke                     # solo tests marcados como smoke
pytest --cov=src --cov-report=html  # coverage + reporte HTML
pytest --tb=short                   # traceback corto al fallar
pytest -x                           # parar al primer fallo
pytest --lf                         # solo los que fallaron última vez
✅ Regla del equipo: cobertura ≥ 80% en lógica de negocio. No buscar el 100% — testear los caminos importantes, no las getters/setters.
