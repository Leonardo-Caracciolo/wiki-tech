Tipado estático — Type Hints

Desde Python 3.10+, el tipado es expresivo y no requiere importar todo desde typing. Usarlo hace el código más legible y permite que los IDEs detecten errores antes de ejecutar.

# Python 3.10+ — sintaxis moderna
from __future__ import annotations

def procesar_clientes(
    clientes:   list[dict[str, str]],
    activo:     bool = True,
    limite:     int | None = None,
) -> list[str]:
    """Filtra y devuelve nombres de clientes activos."""
    resultado = [c["nombre"] for c in clientes if c.get("activo") == activo]
    return resultado[:limite] if limite else resultado

# Dataclasses con tipado — preferir sobre dicts
from dataclasses import dataclass, field
from datetime import date

@dataclass
class Cliente:
    id:       int
    nombre:   str
    cuit:     str
    activo:   bool = True
    tags:     list[str] = field(default_factory=list)

# Pydantic — validación en runtime (ideal para APIs y config)
from pydantic import BaseModel, Field

class ClienteSchema(BaseModel):
    id:     int
    nombre: str = Field(min_length=1, max_length=200)
    cuit:   str = Field(pattern=r"^\d{2}-\d{8}-\d$")
    activo: bool = True

Docstrings — Formato Google Style Recomendado

El stack recomendado es docstrings Google-style en el código → mkdocstrings los lee → MkDocs genera el sitio automáticamente.

def calcular_margen(
    precio_venta: float,
    costo:        float,
    descuento:    float = 0.0,
) -> float:
    """Calcula el margen bruto sobre el precio de venta.

    Aplica el descuento antes de calcular el margen. Si el precio
    de venta es cero, retorna 0.0 para evitar división por cero.

    Args:
        precio_venta: Precio final de venta sin impuestos.
        costo:        Costo directo del producto o servicio.
        descuento:    Descuento porcentual (0.0 a 1.0). Default 0.0.

    Returns:
        Margen bruto como proporción (0.0 a 1.0).
        Ejemplo: 0.35 = 35% de margen.

    Raises:
        ValueError: Si descuento no está entre 0.0 y 1.0.

    Example:
        >>> calcular_margen(100.0, 65.0)
        0.35
        >>> calcular_margen(100.0, 65.0, descuento=0.1)
        0.2778
    """
    if not (0.0 <= descuento <= 1.0):
        raise ValueError(f"descuento debe estar entre 0.0 y 1.0, got {descuento}")
    precio_efectivo = precio_venta * (1 - descuento)
    if precio_efectivo == 0:
        return 0.0
    return (precio_efectivo - costo) / precio_efectivo

# Excepciones personalizadas — siempre definirlas
class BPSError(Exception):
    """Error base del proyecto."""

class ValidationError(BPSError):
    """Error de validación de datos de negocio."""

class ExternalServiceError(BPSError):
    """Error al conectar con un servicio externo (ARCA, SharePoint, etc.)."""

# Uso
def procesar(data: dict) -> dict:
    logger.info("Iniciando procesamiento", extra={"id": data.get("id")})
    try:
        resultado = _procesar_interno(data)
        logger.info("Procesamiento exitoso")
        return resultado
    except ValidationError as e:
        logger.warning(f"Dato inválido: {e}")
        raise
    except Exception as e:
        logger.error(f"Error inesperado: {e}", exc_info=True)
        raise BPSError("Error interno") from e

# ── Context Manager para archivos Excel ──────────────────────
from contextlib import contextmanager
import openpyxl

@contextmanager
def abrir_excel(path: str, read_only: bool = False):
    wb = openpyxl.load_workbook(path, read_only=read_only)
    try:
        yield wb
    finally:
        wb.close()

# Uso limpio
with abrir_excel("datos.xlsx") as wb:
    ws = wb.active
    datos = [[c.value for c in row] for row in ws.iter_rows(min_row=2)]

# ── Retry con backoff (para servicios externos) ───────────────
import time
from functools import wraps

def retry(max_attempts: int = 3, delay: float = 1.0):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1:
                        raise
                    time.sleep(delay * (2 ** attempt))
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2.0)
