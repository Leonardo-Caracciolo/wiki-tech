pytest — Template base completo del equipo

Template real extraído de los proyectos del equipo BPS. Copiar en cada proyecto nuevo bajo tests/unit/.

"""
Template base para tests unitarios con pytest — BPS Dev Team.

Convenciones:
  - Un archivo test_módulo.py por módulo a testear.
  - Una clase TestXxx por clase o función principal.
  - Método: test_<caso>_<condición>_<resultado_esperado>
  - Fixtures compartidas en conftest.py
  - Mocks para TODAS las dependencias externas (ARCA, Excel, APIs).
"""
import pytest
from unittest.mock import MagicMock, patch
from datetime import date

from src.project_name.invoice_processor import (
    Invoice,
    validar_comprobante,
    descargar_comprobantes,
)


# ── FIXTURES ──────────────────────────────────────────────────────────────────
# Las fixtures compartidas van en conftest.py
# Acá como ejemplo local para este test

@pytest.fixture
def factura_valida() -> Invoice:
    """Comprobante válido para happy path tests."""
    return Invoice(
        cuit_emisor="20123456789",
        numero="0001-00001234",
        fecha=date(2024, 3, 1),
        importe_total=12100.00,
        cae="74123456789012",
        tipo="A",
    )

@pytest.fixture
def cuit_receptor_valido() -> str:
    return "30987654321"


# ── TESTS: ValidarComprobante ─────────────────────────────────────────────────

class TestValidarComprobante:
    """Tests para la función validar_comprobante."""

    def test_comprobante_valido_retorna_true(self, factura_valida, cuit_receptor_valido):
        """Happy path: comprobante con todos los campos correctos."""
        resultado = validar_comprobante(factura_valida, cuit_receptor_valido)
        assert resultado is True

    def test_importe_negativo_retorna_false(self, factura_valida, cuit_receptor_valido):
        """Importe negativo no es un comprobante válido."""
        factura_valida.importe_total = -100.0
        resultado = validar_comprobante(factura_valida, cuit_receptor_valido)
        assert resultado is False

    def test_cuit_receptor_invalido_lanza_value_error(self, factura_valida):
        """CUIT con longitud incorrecta debe levantar ValueError."""
        with pytest.raises(ValueError, match="11 dígitos"):
            validar_comprobante(factura_valida, "123")

    def test_none_como_invoice_lanza_type_error(self, cuit_receptor_valido):
        """Pasar None como invoice debe levantar TypeError."""
        with pytest.raises(TypeError):
            validar_comprobante(None, cuit_receptor_valido)

    @pytest.mark.parametrize("importe", [0.0, -1.0, -999.99])
    def test_importes_invalidos_retornan_false(self, factura_valida, cuit_receptor_valido, importe):
        """Parametrize: mismo test para múltiples valores de importe inválido."""
        factura_valida.importe_total = importe
        assert validar_comprobante(factura_valida, cuit_receptor_valido) is False


# ── TESTS: DescargarComprobantes — con mocks ──────────────────────────────────

class TestDescargarComprobantes:
    """Tests para descargar_comprobantes — mockea la conexión a ARCA."""

    @patch("src.project_name.invoice_processor._conectar_arca")
    def test_retorna_lista_vacia_si_no_hay_comprobantes(self, mock_arca):
        """Si ARCA devuelve vacío, la función retorna lista vacía sin error."""
        mock_arca.return_value.consultar.return_value = []
        resultado = descargar_comprobantes("20123456789", date(2024,1,1), date(2024,1,31))
        assert resultado == []

    @patch("src.project_name.invoice_processor._conectar_arca")
    def test_error_conexion_se_propaga_como_connection_error(self, mock_arca):
        """Si ARCA no responde, debe propagarse ConnectionError al caller."""
        mock_arca.side_effect = ConnectionError("Timeout ARCA")
        with pytest.raises(ConnectionError):
            descargar_comprobantes("20123456789", date(2024,1,1), date(2024,1,31))

conftest.py — Fixtures compartidas entre todos los tests

"""
conftest.py — Fixtures compartidas entre todos los tests del proyecto.
pytest las descubre automáticamente en tests/. No necesitan ser importadas.
"""
import pytest
from datetime import date
from unittest.mock import MagicMock
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker

from src.project_name.invoice_processor import Invoice


@pytest.fixture(scope="session")
def arca_mock():
    """Mock del cliente ARCA — reutilizable en todos los tests."""
    mock = MagicMock()
    mock.consultar.return_value = [
        Invoice("20123456789", "0001-00000001", date(2024,1,5), 12100.0, "74000000000001", "A"),
        Invoice("20123456789", "0001-00000002", date(2024,1,12), 5500.0,  "74000000000002", "B"),
    ]
    return mock


@pytest.fixture
def factura_valida():
    """Comprobante válido para happy path. Reset por cada test."""
    return Invoice(
        cuit_emisor="20123456789",
        numero="0001-00001234",
        fecha=date(2024, 3, 1),
        importe_total=12100.00,
        cae="74123456789012",
        tipo="A",
    )


@pytest.fixture(scope="session")
def engine():
    """SQLite en memoria — cero setup, tests rápidos."""
    return create_engine("sqlite:///:memory:")


@pytest.fixture(scope="function")
def db_session(engine):
    """Sesión con rollback automático — cada test arranca con DB limpia."""
    connection = engine.connect()
    transaction = connection.begin()
    Session = sessionmaker(bind=connection)
    session = Session()
    yield session
    session.close()
    transaction.rollback()
    connection.close()

Smoke Tests — verificación post-deploy

Suite mínima que verifica que el sistema está vivo. Se corre después de cada deploy a producción. Máximo 30 segundos en total.

# tests/smoke/test_smoke.py
"""
Smoke tests — verificación rápida post-deploy.
Reglas:
  - Máximo 30 segundos de ejecución total
  - NO modifican datos de producción
  - NO fallan por datos variables (fecha, monto, etc.)
  - Solo verifican que el sistema responde y las funciones críticas existen
"""
import pytest
import requests

BASE_URL = "http://localhost:8000"

@pytest.mark.smoke
def test_api_responde():
    """La API debe estar viva."""
    resp = requests.get(f"{BASE_URL}/health", timeout=5)
    assert resp.status_code == 200

@pytest.mark.smoke
def test_base_de_datos_accesible():
    """La DB debe responder al health check."""
    resp = requests.get(f"{BASE_URL}/health/db", timeout=5)
    assert resp.json()["status"] == "ok"

@pytest.mark.smoke
def test_endpoint_critico_responde():
    """El endpoint principal del proceso debe responder."""
    resp = requests.get(f"{BASE_URL}/api/comprobantes", timeout=10)
    assert resp.status_code in (200, 401)  # OK o requiere auth — ambos = "vivo"

@pytest.mark.smoke
def test_version_es_la_esperada():
    """La versión desplegada debe coincidir con la versión del release."""
    resp = requests.get(f"{BASE_URL}/health", timeout=5)
    data = resp.json()
    assert "version" in data

Testing avanzado — Mocking, parametrización y property-based

from unittest.mock import MagicMock, patch, call
from hypothesis   import given, strategies as st

# ── Verificar llamadas con argumentos específicos ──────────────────────────────
@patch("mymodule.ArcaClient")
def test_cliente_arca_recibe_cuit_correcto(mock_client):
    mock_instance = mock_client.return_value
    mock_instance.consultar.return_value = []

    descargar_comprobantes("20123456789", date(2024,1,1), date(2024,1,31))

    mock_instance.consultar.assert_called_once_with(
        cuit="20123456789",
        fecha_desde=date(2024,1,1),
        fecha_hasta=date(2024,1,31)
    )

# ── Mutation testing con hypothesis ───────────────────────────────────────────
@given(st.floats(min_value=0.01, max_value=10_000_000))
def test_margen_nunca_supera_uno_con_datos_aleatorios(precio):
    margen = calcular_margen(precio, precio * 0.6)
    assert 0 <= margen <= 1

# ── Test doubles — Mock vs Stub vs Fake ───────────────────────────────────────
# Stub: retorna valor fijo — "qué devuelve"
mock_arca = MagicMock()
mock_arca.consultar.return_value = ["factura_1", "factura_2"]

# Mock: también verifica llamadas — "cómo se usa"
mock_arca.consultar.assert_called_once()
mock_arca.consultar.assert_called_with(cuit="20123456789")

# Spy — capturar argumentos para inspeccionarlos
args_capturados = []
mock_arca.consultar.side_effect = lambda **kw: args_capturados.append(kw) or []

Comandos esenciales de pytest

pytest                               # todos los tests
pytest -v                            # verbose — nombre completo de cada test
pytest -k "validar"                 # solo tests con "validar" en el nombre
pytest -m smoke                      # solo tests marcados como @pytest.mark.smoke
pytest --cov=src --cov-report=html   # coverage + reporte HTML en htmlcov/
pytest --cov=src --cov-report=term   # coverage en terminal
pytest --tb=short                    # traceback corto al fallar (más legible)
pytest -x                            # parar al primer fallo
pytest --lf                          # solo los tests que fallaron la última vez
pytest -n 4                          # correr en paralelo con 4 workers (pytest-xdist)
pytest tests/smoke/ -m smoke         # correr solo la suite de smoke

# Robot Framework
robot tests/e2e/                     # todos los tests E2E
robot --include smoke tests/          # solo tests con tag smoke
robot --output results/ tests/        # guardar resultados en results/
robot --variable ENV:staging tests/  # pasar variable de entorno
✅ Regla del equipo: cobertura ≥ 80% en lógica de negocio. No buscar el 100% — testear los caminos críticos, los casos borde y los errores. Las getters/setters triviales no necesitan test.
