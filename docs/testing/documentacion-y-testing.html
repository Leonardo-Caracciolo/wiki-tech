Módulo para procesamiento y validación de comprobantes ARCA.

Este módulo provee utilidades para descargar, parsear y validar
comprobantes electrónicos obtenidos mediante los web services de ARCA.
"""
from dataclasses import dataclass
from datetime import date
from typing import Optional


@dataclass
class Invoice:
    """Representa un comprobante electrónico de ARCA.

    Attributes:
        cuit_emisor: CUIT del emisor sin guiones (11 dígitos).
        numero: Número de comprobante (punto de venta + número).
        fecha: Fecha de emisión del comprobante.
        importe_total: Importe total incluyendo impuestos.
        cae: Código de Autorización Electrónico otorgado por ARCA.
        tipo: Tipo de comprobante (A, B, C, M, E).
    """
    cuit_emisor: str
    numero: str
    fecha: date
    importe_total: float
    cae: str
    tipo: str


def validar_comprobante(invoice: Invoice, cuit_receptor: str) -> bool:
    """Valida un comprobante electrónico contra los criterios del receptor.

    Verifica que el comprobante sea del tipo correcto según la condición
    fiscal del receptor y que el CAE sea formalmente válido (no verifica
    contra ARCA, solo formato).

    Args:
        invoice: Instancia del comprobante a validar.
        cuit_receptor: CUIT del receptor sin guiones (11 dígitos).

    Returns:
        True si el comprobante pasa todas las validaciones, False en caso
        contrario.

    Raises:
        ValueError: Si el CUIT del receptor no tiene 11 dígitos.
        TypeError: Si invoice no es una instancia de Invoice.

    Example:
        >>> from datetime import date
        >>> inv = Invoice(
        ...     cuit_emisor="20123456789",
        ...     numero="0001-00001234",
        ...     fecha=date(2024, 3, 1),
        ...     importe_total=12100.00,
        ...     cae="74123456789012",
        ...     tipo="A"
        ... )
        >>> validar_comprobante(inv, "30987654321")
        True
    """
    if not isinstance(invoice, Invoice):
        raise TypeError(f"Se esperaba Invoice, se recibió {type(invoice).__name__}")
    if len(cuit_receptor.replace("-", "")) != 11:
        raise ValueError(f"CUIT inválido: {cuit_receptor!r}. Debe tener 11 dígitos.")

    # Responsable Inscripto debe recibir comprobante tipo A
    if len(invoice.cae) != 14:
        return False
    if invoice.importe_total <= 0:
        return False

    return True

' ModuloComprobantes.bas
' Procesamiento de comprobantes para reportes de Transfer Pricing
'
'@ModuleDescription("Módulo para lectura y validación de comprobantes desde archivos Excel.")
'@Folder("Comprobantes")

Option Explicit

'@Description("Representa un comprobante fiscal con sus atributos principales.")
Public Type TComprobante
    CUIT As String
    Numero As String
    Fecha As Date
    ImporteTotal As Double
    TipoComprobante As String
End Type


'@Description("Valida que un comprobante tenga todos los campos requeridos y valores dentro del rango esperado.")
'@Param comprobante "Estructura TComprobante a validar."
'@Param cuitReceptor "CUIT del receptor sin guiones (11 dígitos)."
'@Returns "True si el comprobante es válido, False en caso contrario."
'@Remarks "No verifica el CAE contra ARCA, solo valida formato y valores."
Public Function ValidarComprobante(ByVal comprobante As TComprobante, _
                                    ByVal cuitReceptor As String) As Boolean
    ValidarComprobante = False

    ' Validar CUIT receptor
    If Len(Replace(cuitReceptor, "-", "")) <> 11 Then
        Debug.Print "ERROR: CUIT receptor inválido: " & cuitReceptor
        Exit Function
    End If

    ' Validar campos obligatorios
    If Len(Trim(comprobante.CUIT)) = 0 Then Exit Function
    If Len(Trim(comprobante.Numero)) = 0 Then Exit Function
    If comprobante.ImporteTotal <= 0 Then Exit Function
    If comprobante.Fecha = #1/1/1900# Then Exit Function

    ValidarComprobante = True
End Function


'@Description("Lee todos los comprobantes del rango activo en la hoja especificada.")
'@Param nombreHoja "Nombre de la hoja de Excel que contiene los datos."
'@Param filaInicio "Número de fila donde comienzan los datos (sin encabezado)."
'@Returns "Array de TComprobante con los registros leídos."
'@Error "Lanza error 9 si la hoja no existe."
'@Example
'   Dim comprobantes() As TComprobante
'   comprobantes = LeerComprobantes("Facturas", 2)
'   Debug.Print "Total leídos: " & UBound(comprobantes) + 1
Public Function LeerComprobantes(ByVal nombreHoja As String, _
                                  ByVal filaInicio As Long) As TComprobante()
    Dim ws As Worksheet
    On Error GoTo ErrorHandler

    Set ws = ThisWorkbook.Worksheets(nombreHoja)

    Dim ultimaFila As Long
    ultimaFila = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    Dim resultado() As TComprobante
    Dim count As Long
    count = 0

    ReDim resultado(ultimaFila - filaInicio)

    Dim i As Long
    For i = filaInicio To ultimaFila
        resultado(count).CUIT = CStr(ws.Cells(i, 1).Value)
        resultado(count).Numero = CStr(ws.Cells(i, 2).Value)
        resultado(count).Fecha = CDate(ws.Cells(i, 3).Value)
        resultado(count).ImporteTotal = CDbl(ws.Cells(i, 4).Value)
        resultado(count).TipoComprobante = CStr(ws.Cells(i, 5).Value)
        count = count + 1
    Next i

    LeerComprobantes = resultado
    Exit Function

ErrorHandler:
    If Err.Number = 9 Then
        Err.Raise 9, "LeerComprobantes", "La hoja '" & nombreHoja & "' no existe."
    End If
    Err.Raise Err.Number, Err.Source, Err.Description
End Function

using System;
using System.Collections.Generic;

namespace RpaTeam.Comprobantes
{
    /// <summary>
    /// Representa un comprobante electrónico emitido a través de ARCA.
    /// </summary>
    /// <remarks>
    /// Esta clase es inmutable una vez creada. Para modificar datos,
    /// crear una nueva instancia con los valores actualizados.
    /// </remarks>
    public class Invoice
    {
        /// <summary>Obtiene el CUIT del emisor (11 dígitos, sin guiones).</summary>
        public string CuitEmisor { get; }

        /// <summary>Obtiene el número del comprobante en formato PPPP-NNNNNNNN.</summary>
        public string Numero { get; }

        /// <summary>Obtiene la fecha de emisión del comprobante.</summary>
        public DateTime Fecha { get; }

        /// <summary>Obtiene el importe total incluyendo impuestos.</summary>
        public decimal ImporteTotal { get; }

        /// <summary>Inicializa una nueva instancia de <see cref="Invoice"/>.</summary>
        /// <param name="cuitEmisor">CUIT del emisor sin guiones (11 dígitos).</param>
        /// <param name="numero">Número de comprobante en formato PPPP-NNNNNNNN.</param>
        /// <param name="fecha">Fecha de emisión.</param>
        /// <param name="importeTotal">Importe total con impuestos incluidos.</param>
        /// <exception cref="ArgumentNullException">
        ///   Si <paramref name="cuitEmisor"/> o <paramref name="numero"/> son null.
        /// </exception>
        /// <exception cref="ArgumentOutOfRangeException">
        ///   Si <paramref name="importeTotal"/> es menor o igual a cero.
        /// </exception>
        public Invoice(string cuitEmisor, string numero, DateTime fecha, decimal importeTotal)
        {
            CuitEmisor = cuitEmisor ?? throw new ArgumentNullException(nameof(cuitEmisor));
            Numero = numero ?? throw new ArgumentNullException(nameof(numero));
            Fecha = fecha;
            ImporteTotal = importeTotal > 0
                ? importeTotal
                : throw new ArgumentOutOfRangeException(nameof(importeTotal), "Debe ser mayor a cero.");
        }
    }

    /// <summary>
    /// Provee métodos de validación para comprobantes electrónicos.
    /// </summary>
    public static class InvoiceValidator
    {
        /// <summary>
        /// Valida que un comprobante cumpla con los criterios formales del receptor.
        /// </summary>
        /// <param name="invoice">El comprobante a validar.</param>
        /// <param name="cuitReceptor">CUIT del receptor sin guiones (11 dígitos).</param>
        /// <returns>
        ///   <see langword="true"/> si el comprobante pasa todas las validaciones;
        ///   <see langword="false"/> en caso contrario.
        /// </returns>
        /// <exception cref="ArgumentNullException">
        ///   Si <paramref name="invoice"/> o <paramref name="cuitReceptor"/> son null.
        /// </exception>
        /// <exception cref="ArgumentException">
        ///   Si <paramref name="cuitReceptor"/> no tiene exactamente 11 dígitos.
        /// </exception>
        /// <example>
        /// <code>
        /// var inv = new Invoice("20123456789", "0001-00001234", DateTime.Now, 12100m);
        /// bool esValido = InvoiceValidator.Validar(inv, "30987654321");
        /// Console.WriteLine(esValido); // True
        /// </code>
        /// </example>
        public static bool Validar(Invoice invoice, string cuitReceptor)
        {
            if (invoice is null) throw new ArgumentNullException(nameof(invoice));
            if (cuitReceptor is null) throw new ArgumentNullException(nameof(cuitReceptor));

            var cuitLimpio = cuitReceptor.Replace("-", "");
            if (cuitLimpio.Length != 11)
                throw new ArgumentException($"CUIT inválido: '{cuitReceptor}'. Debe tener 11 dígitos.", nameof(cuitReceptor));

            return invoice.ImporteTotal > 0 && invoice.Fecha <= DateTime.Today;
        }
    }
}

"""
Template base para tests unitarios con pytest.

Convenciones:
  - Un archivo test_ por módulo a testear.
  - Una clase TestXxx por clase/función principal.
  - Método: test_<caso>_<condicion>_<resultado_esperado>
  - Fixtures compartidas en conftest.py
  - Mocks para todas las dependencias externas.
"""
import pytest
from unittest.mock import MagicMock, patch
from datetime import date

# Importar el módulo bajo test
from src.project_name.invoice_processor import (
    Invoice,
    validar_comprobante,
    descargar_comprobantes,
)


# ── FIXTURES ─────────────────────────────────────────────────────────────────
# Las fixtures van en conftest.py para reutilizarlas entre archivos.
# Acá como ejemplo local.

@pytest.fixture
def factura_valida() -> Invoice:
    """Comprobante válido para usar en tests de happy path."""
    return Invoice(
        cuit_emisor="20123456789",
        numero="0001-00001234",
        fecha=date(2024, 3, 1),
        importe_total=12100.00,
        cae="74123456789012",
        tipo="A",
    )

@pytest.fixture
def cuit_receptor_valido() -> str:
    return "30987654321"


# ── TESTS: ValidarComprobante ─────────────────────────────────────────────────

class TestValidarComprobante:
    """Tests para la función validar_comprobante."""

    def test_comprobante_valido_retorna_true(self, factura_valida, cuit_receptor_valido):
        """Happy path: comprobante con todos los campos correctos."""
        resultado = validar_comprobante(factura_valida, cuit_receptor_valido)
        assert resultado is True

    def test_importe_negativo_retorna_false(self, factura_valida, cuit_receptor_valido):
        """Importe negativo no es un comprobante válido."""
        factura_valida.importe_total = -100.0
        resultado = validar_comprobante(factura_valida, cuit_receptor_valido)
        assert resultado is False

    def test_cuit_receptor_con_menos_de_11_digitos_lanza_error(self, factura_valida):
        """CUIT con longitud incorrecta debe levantar ValueError."""
        with pytest.raises(ValueError, match="11 dígitos"):
            validar_comprobante(factura_valida, "123")

    def test_invoice_none_lanza_type_error(self, cuit_receptor_valido):
        """Pasar None como invoice debe levantar TypeError."""
        with pytest.raises(TypeError):
            validar_comprobante(None, cuit_receptor_valido)

    @pytest.mark.parametrize("importe", [0.0, -1.0, -999.99])
    def test_importes_invalidos_retornan_false(self, factura_valida, cuit_receptor_valido, importe):
        """Importes menores o iguales a cero deben fallar validación."""
        factura_valida.importe_total = importe
        assert validar_comprobante(factura_valida, cuit_receptor_valido) is False

class TestDescargarComprobantes:

"""
Tests de integración VBA ↔ Python usando xlwings.
Requiere Excel instalado en la máquina que corre el CI.

Instalar: pip install xlwings
"""
import pytest
import xlwings as xw
from pathlib import Path

EXCEL_PATH = Path("data/ReporteTransferPricing.xlsm")


@pytest.fixture(scope="module")
def excel_app():
    """Abre el Excel en background sin interfaz gráfica."""
    app = xw.App(visible=False, add_book=False)
    wb = app.books.open(str(EXCEL_PATH.resolve()))
    yield wb
    wb.close()
    app.quit()


class TestVBAMacros:
    """Tests que invocan macros VBA desde Python y verifican el resultado."""

    def test_generar_reporte_crea_hoja_resultado(self, excel_app):
        """La macro GenerarReporte debe crear la hoja 'Resultado'."""
        wb = excel_app

        # Limpiar estado previo
        try:
            wb.sheets["Resultado"].delete()
        except Exception:
            pass

        # Ejecutar la macro VBA
        wb.macro("ModuloReporte.GenerarReporte")()

        # Verificar que la hoja fue creada
        sheet_names = [s.name for s in wb.sheets]
        assert "Resultado" in sheet_names, "La macro debe crear la hoja 'Resultado'."

    def test_reporte_tiene_encabezados_correctos(self, excel_app):
        """La hoja Resultado debe tener los encabezados definidos en el PDD."""
        sheet = excel_app.sheets["Resultado"]
        encabezados_esperados = ["CUIT", "Razón Social", "Importe", "Fecha", "CAE"]

        for i, header in enumerate(encabezados_esperados, start=1):
            valor = sheet.cells(1, i).value
            assert valor == header, f"Col {i}: se esperaba '{header}', se encontró '{valor}'."

using System;
using Xunit;
using Moq;
using FluentAssertions;
using RpaTeam.Comprobantes;

namespace RpaTeam.Tests.Comprobantes
{
    /// <summary>
    /// Tests para InvoiceValidator.
    /// Convención: Método_Condición_ResultadoEsperado
    /// </summary>
    public class InvoiceValidatorTests
    {
        // ── FIXTURES (shared setup) ──────────────────────────────────────
        private static Invoice CreateValidInvoice() => new Invoice(
            cuitEmisor: "20123456789",
            numero: "0001-00001234",
            fecha: new DateTime(2024, 3, 1),
            importeTotal: 12100m
        );

        // ── TESTS: Happy Path ────────────────────────────────────────────

        [Fact]
        public void Validar_FacturaValida_RetornaTrue()
        {
            // Arrange
            var invoice = CreateValidInvoice();

            // Act
            var resultado = InvoiceValidator.Validar(invoice, "30987654321");

            // Assert
            resultado.Should().BeTrue("un comprobante con datos válidos debe pasar la validación.");
        }

        // ── TESTS: Validación de argumentos ─────────────────────────────

        [Fact]
        public void Validar_InvoiceNull_LanzaArgumentNullException()
        {
            // Act
            Action accion = () => InvoiceValidator.Validar(null!, "30987654321");

            // Assert
            accion.Should().Throw<ArgumentNullException>()
                .WithParameterName("invoice");
        }

        [Fact]
        public void Validar_CuitReceptorNull_LanzaArgumentNullException()
        {
            var invoice = CreateValidInvoice();
            Action accion = () => InvoiceValidator.Validar(invoice, null!);
            accion.Should().Throw<ArgumentNullException>()
                .WithParameterName("cuitReceptor");
        }

        [Theory]
        [InlineData("123")]
        [InlineData("")]
        [InlineData("1234567890123")]  // 13 dígitos — muy largo
        public void Validar_CuitReceptorInvalido_LanzaArgumentException(string cuitInvalido)
        {
            var invoice = CreateValidInvoice();
            Action accion = () => InvoiceValidator.Validar(invoice, cuitInvalido);
            accion.Should().Throw<ArgumentException>()
                .WithParameterName("cuitReceptor");
        }

        // ── TESTS: Reglas de negocio ─────────────────────────────────────

        [Theory]
        [InlineData(0)]
        [InlineData(-1)]
        [InlineData(-999.99)]
        public void Validar_ImporteNoPositivo_LanzaArgumentOutOfRange(decimal importe)
        {
            // Los importes inválidos son rechazados en el constructor de Invoice
            Action accion = () => new Invoice("20123456789", "0001-0001", DateTime.Now, importe);
            accion.Should().Throw<ArgumentOutOfRangeException>()
                .WithParameterName("importeTotal");
        }

        [Fact]
        public void Validar_FechaFutura_RetornaFalse()
        {
            var invoice = new Invoice(
                "20123456789", "0001-00001234",
                DateTime.Today.AddDays(1),  // Fecha futura
                12100m
            );
            var resultado = InvoiceValidator.Validar(invoice, "30987654321");
            resultado.Should().BeFalse("un comprobante con fecha futura no es válido.");
        }
    }
}

from datetime import date
from dataclasses import dataclass


@dataclass
class Invoice:
    """Representa un comprobante electrónico.

    Attributes:
        cuit_emisor: CUIT del emisor (11 dígitos).
        importe_total: Importe total con impuestos.
        tipo: Tipo de comprobante (A, B, C).
    """
    cuit_emisor: str
    importe_total: float
    tipo: str


def formatear_importe(importe: float) -> str:
    """Formatea un importe como string con separador de miles y 2 decimales.

    Args:
        importe: Valor numérico a formatear.

    Returns:
        String formateado con $ y separadores de miles.

    Raises:
        ValueError: Si el importe es negativo.

    Example:
        >>> formatear_importe(12100.5)
        '$12.100,50'

        >>> formatear_importe(0)
        '$0,00'

        >>> formatear_importe(-100)
        Traceback (most recent call last):
            ...
        ValueError: El importe no puede ser negativo.
    """
    if importe < 0:
        raise ValueError("El importe no puede ser negativo.")
    entero = int(importe)
    centavos = round((importe - entero) * 100)
    entero_fmt = f"{entero:,}".replace(",", ".")
    return f"${entero_fmt},{centavos:02d}"


def es_comprobante_tipo_a(invoice: Invoice) -> bool:
    """Verifica si un comprobante es de tipo A (Responsable Inscripto).

    Args:
        invoice: Instancia del comprobante a verificar.

    Returns:
        True si el tipo es 'A', False en cualquier otro caso.

    Example:
        >>> inv_a = Invoice("20123456789", 12100.0, "A")
        >>> es_comprobante_tipo_a(inv_a)
        True

        >>> inv_b = Invoice("20123456789", 5500.0, "B")
        >>> es_comprobante_tipo_a(inv_b)
        False
    """
    return invoice.tipo == "A"

Mutation Testing — mutmut & Stryker
Mutation testing modifica el código fuente deliberadamente (cambia > por >=, elimina condiciones) y verifica que los tests fallen. Si un test no detecta la mutación, el test no sirve. Mide la calidad real de los tests, no solo la cobertura.
mutmutStryker (.NET)CI/CD gate
Concepto
Cómo ejecutar — Python (mutmut)
Cómo ejecutar — C# (Stryker)
💡
Cobertura 100% ≠ tests buenos. Un test puede ejecutar una línea sin verificar nada. Mutation testing detecta esos tests vacíos. Un "mutante vivo" es un bug potencial que los tests no detectan.
HerramientaLenguajeQué mutaCI/CD

Cosmic RayPythonMás configurable que mutmutSí

⚠️
Cuándo correrlo: mutation testing es lento (puede tardar horas en proyectos grandes). Correlo en el pipeline de release, no en cada PR. En PRs, solo tests unitarios y de integración.

      echo "Mutation score: $RATE% survived"
      if (( $(echo "$RATE > 20" | bc -l) )); then

Test Doubles — Mock, Stub, Fake, Spy, Dummy
El término "mock" se usa para todo pero hay 5 tipos distintos de test doubles. Usar el tipo correcto hace los tests más simples, más mantenibles y con mensajes de error más claros cuando fallan.
MockStubFakeSpyDummy
Los 5 tipos
Ejemplos — Python
Ejemplos — C#
💡
Definición de Martin Fowler: Test Double es el término genérico para cualquier objeto que reemplaza a uno real durante un test. Los 5 tipos se diferencian por su comportamiento y propósito.
TipoQué haceVerifica llamadasCuándo usarlo
DummyObjeto vacío que nunca se usa realmenteNoCuando el parámetro es requerido pero no importa en el test
StubDevuelve respuestas preconfiguradasNoSimular respuestas de APIs, bases de datos, archivos
FakeImplementación simplificada pero funcionalNoBase de datos en memoria, filesystem falso
SpyStub que también registra cómo fue llamadoSí (a posteriori)Verificar que algo fue llamado N veces con ciertos args
MockStub con expectativas preconfiguradas que fallan si no se cumplenSí (estricto)Cuando el comportamiento de llamada es parte del contrato

Smoke Tests
El conjunto mínimo de tests que verifican que el sistema "prende". Si un smoke test falla, no tiene sentido correr los otros 200 tests — algo fundamental está roto. Son pocos, son rápidos, y cubren solo el happy path crítico de cada funcionalidad principal.

Cómo ejecutar
💡
Origen del nombre: viene de la electrónica. Cuando armás un circuito nuevo, lo enchufás y verificás que no salga humo antes de hacer pruebas más serias. En software: verificás que el sistema no "explota" antes de correr la suite completa.
Características de un buen smoke test
⚡
Rápidos
Deben correr en segundos, no minutos. Si tardan más de 2 minutos en total, hay demasiados o están haciendo demasiado. El objetivo es feedback inmediato.
🎯
Pocos y críticos
Entre 5 y 15 casos máximo por módulo. Solo cubren el happy path de las funcionalidades más importantes. No casos borde, no excepciones, no edge cases.
🚦
Gate del pipeline
Si un smoke test falla, el pipeline se detiene ahí. No tiene sentido correr 200 tests de detalle si lo fundamental no funciona.
🔄
Se corren siempre
En cada commit, en cada PR, en cada deploy. Son el primer paso del CI. Los tests de integración y regresión pueden saltearse; los smokes nunca.

Principio guía: los tests no son una actividad separada que se hace "al final". Se escriben mientras se codea, un test por cada comportamiento que implementás. El CI los corre automáticamente — vos no tenés que acordarte de correrlos.

⚠️
Cuándo NO escribir tests: scripts desechables de una sola corrida, prototipos de exploración (POC), scripts de migración de datos que corren una vez. En esos casos documentar el comportamiento esperado en comentarios es suficiente.
📘
Este paso se hace una sola vez por feature, antes de escribir una línea de código de producción. Toma 15-30 minutos y evita reescribir tests después.
1
Leer los criterios de aceptación del PDD
Identificar cada comportamiento que el código debe tener. Cada criterio del PDD se convierte en al menos un test. Pregunta clave: ¿cómo voy a saber que esto funciona?
2
Listar los casos a testear antes de codear
Escribir en comentarios la lista de casos: happy path, errores esperados, casos borde. Ejemplo: # TODO test: importe negativo → False, # TODO test: CUIT vacío → ValueError. No el código del test, solo la lista.
3
Crear el archivo de test vacío
Crear tests/unit/test_nombre_modulo.py con la estructura de clases vacía. Definir las clases TestNombreClase y los métodos vacíos con nombres descriptivos. Esto fuerza a pensar en la API antes de implementarla.
4
Configurar fixtures en conftest.py
Identificar qué datos de prueba se van a reutilizar y crear las fixtures compartidas en tests/conftest.py. Una fixture por entidad de negocio (factura válida, factura inválida, CUIT válido, etc.).
5
Verificar que el entorno funciona
Correr pytest --collect-only para confirmar que pytest descubre los archivos. Si no aparecen, hay un problema de estructura de carpetas o nombre de archivo.

💡
Ciclo por cada función que implementás: (1) escribir el test que describe el comportamiento esperado → (2) correr el test, debe fallar → (3) implementar la función → (4) correr el test, debe pasar → (5) refactorizar si hace falta. Este es el ciclo TDD, pero podés hacerlo en cualquier orden siempre que el test exista antes del merge.
1
Escribir el test del happy path primero
Antes de implementar la función, escribir el test que describe cómo debería funcionar en el caso normal. Si no podés escribir el test, es porque el diseño de la función no está claro — resolverlo antes de codear.
2
Correr el test — debe fallar (rojo)
Ejecutar pytest tests/unit/test_modulo.py::TestClase::test_funcion -v. Si pasa sin haber implementado nada, el test está mal escrito o la función ya existe. Si falla con el error esperado, todo está bien.
3
Implementar la función hasta que el test pase
Escribir el código mínimo para que el test pase. No agregar lógica extra todavía. Correr el test frecuentemente mientras implementás — feedback inmediato.
4
Agregar tests de casos de error
Con el happy path verde, agregar tests para cada excepción documentada en el docstring: pytest.raises(ValueError), pytest.raises(TypeError). Un test por tipo de error.
5
Agregar tests parametrizados para los casos borde
Si hay múltiples valores de entrada que deben producir el mismo tipo de resultado, usar @pytest.mark.parametrize. Evitar copiar y pegar el mismo test cambiando solo un valor.
6
Verificar cobertura del módulo que estás escribiendo
Correr pytest tests/unit/test_modulo.py --cov=src/project/modulo --cov-report=term-missing. Ver qué líneas no están cubiertas. Si hay lógica importante sin cubrir, agregar el test faltante.
7
Correr todos los unitarios antes de commitear
Antes de cada commit: pytest tests/unit/ -q. El pre-commit hook lo hace automáticamente pero es bueno tener el hábito. Si algo falla, corregirlo antes del commit.

Al abrir el PR el CI corre automáticamente. Pero antes de abrirlo, hay un checklist que el dev debe completar localmente para no "romper" el CI del equipo.
1
Correr el smoke suite completo localmente
Antes de abrir el PR: pytest -m smoke -v. Deben pasar todos. Si alguno falla, hay algo roto a nivel fundamental que hay que resolver primero.
2
Correr todos los tests unitarios
pytest tests/unit/ -v. Todos deben estar en verde. No abrir un PR con tests en rojo — el CI va a fallar y bloquea al resto del equipo.
3
Verificar cobertura general del proyecto
pytest tests/unit/ --cov=src --cov-report=term-missing --cov-fail-under=70. Si la cobertura cae por debajo del 70%, agregar los tests que faltan antes de abrir el PR.
4
Correr pre-commit manualmente
pre-commit run --all-files. Esto corre black, flake8, isort, y todos los hooks configurados. El CI corre exactamente lo mismo — si pasa localmente, pasa en el CI.
5
Revisar que los nuevos tests tienen nombres descriptivos
Los nombres de los tests son la documentación de qué hace el código. test_validar_comprobante_importe_negativo_retorna_false es mucho mejor que test_caso2. Revisar todos los tests nuevos del PR.
6
Abrir el PR y esperar el CI
El CI corre smoke → unitarios → cobertura → calidad. Si algo falla, el PR tiene una X roja. Leer el log del CI para entender qué falló. No mergear con CI rojo bajo ninguna circunstancia.

El QA final con Robot Framework se hace en los días 8-9 del sprint contra el entorno de homologación — nunca contra producción. El objetivo es validar el proceso completo, no funciones individuales.
1
Preparar el entorno de homologación
Verificar que el bot está deployado en homologación con el código del sprint actual. Confirmar acceso a los sistemas externos (ARCA, Excel, DB) desde ese entorno. Si no hay acceso, usar stubs configurados en el variables.resource.
2
Revisar los criterios de aceptación del PDD
Los criterios del PDD son los test cases de Robot Framework. Cada criterio se convierte en un TC. Si el PDD dice 'el importe debe aparecer con separador de miles', eso es un test case. Leer el PDD antes de escribir las suites.
3
Escribir los test cases en Robot Framework
Un archivo por cada módulo funcional del proceso: 01_happy_path.robot, 02_excepciones.robot, 03_regression.robot. Los keywords de implementación van en resources/keywords.resource, no inline en los test cases.
4
Correr el smoke de Robot Framework primero
robot --include smoke tests/e2e/robot/suites/. Si el smoke falla, no correr el resto. Hay algo roto a nivel de conectividad o configuración del entorno que hay que resolver primero.
5
Correr la suite completa y revisar el reporte
robot tests/e2e/robot/suites/. Abrir el log.html generado. Revisar cada test que falló: leer el mensaje de error, identificar si es un bug del bot o un problema del entorno de test.
6
Registrar bugs con evidencia
Por cada test que falla: captura de pantalla del log.html, descripción del comportamiento esperado vs. el actual, prioridad (P1/P2/P3), y crear el ticket en Azure DevOps con la referencia al test case.
7
Fix → PR → CI → volver a correr Robot Framework
Los fixes de P1 van en una rama bug/xxx. PR → CI verde → merge a develop → re-correr la suite de Robot Framework completa para verificar que el fix resolvió el problema y no rompió nada más.
8
Validar campo por campo contra el PDD
Para cada output del proceso (Excel, email, DB), validar campo por campo que los valores son correctos según los criterios del PDD. No solo que el proceso terminó sin error — que el resultado es correcto.

Descargar el instalador desde rubberduckvba.com. Cerrar Excel antes de instalar. Al reabrir Excel, aparece el menú Rubberduck en la barra de herramientas. Primera vez: Rubberduck → Refresh para que indexe el proyecto.

Crear el módulo de tests en el VBE
En el VBE (Alt+F11): Insertar → Módulo. Nombrar el módulo TestNombreModulo. Agregar al inicio: '@TestModule, '@Folder("Tests") y Option Explicit. Rubberduck detecta automáticamente que es un módulo de tests.

Agregar las anotaciones de setup

💡
En VBA se sigue el mismo patrón Arrange / Act / Assert que en cualquier otro lenguaje. Arrange: preparar los datos. Act: llamar a la función. Assert: verificar el resultado.
1
Identificar qué funciones testear
Testear todas las funciones Public Function y Public Sub que tienen lógica de negocio: validaciones, cálculos, transformaciones. No testear funciones que solo muestran MsgBox o mueven el cursor.
2
Escribir el test del happy path
Por cada función pública, escribir al menos un test con datos válidos que deben retornar el resultado esperado. Nombre del método: NombreFuncion_Condicion_ResultadoEsperado.
3
Escribir tests de casos de error
Testear qué pasa cuando los datos son inválidos. Para funciones que deben lanzar error, usar On Error Resume Next y verificar Err.Number. Para funciones que retornan False en error, verificar el retorno.
4
Aislar dependencias de Excel
Las funciones que leen celdas son difíciles de testear porque dependen de una hoja. La solución es separar la lógica: una función lee los datos de la hoja, otra función procesa los datos. Testear solo la función de procesamiento.

Public Sub ValidarComprobante_DatosValidos_RetornaTrue()

    comp.Numero = "0001-00001234"

    comp.TipoComprobante = "A"
    
    ' Act
    Dim resultado As Boolean
    resultado = ValidarComprobante(comp, "30987654321")
    
    ' Assert

End Sub

Los tests en Rubberduck se corren desde el Test Explorer dentro del VBE. Es un panel que muestra todos los tests del proyecto con íconos verde/rojo.
1
Abrir el Test Explorer de Rubberduck
En el VBE: Rubberduck → Test Explorer. Se abre un panel con todos los módulos y tests del proyecto. Si no aparece algún test, hacer Rubberduck → Refresh para reindexar el proyecto.
2
Correr todos los tests del módulo
Click derecho sobre el módulo → Run. O seleccionar todos (Ctrl+A) y click en el botón Run. Los tests verdes pasaron, los rojos fallaron. El mensaje de error aparece en el panel inferior.
3
Leer el error de un test que falla
Click sobre el test en rojo. El panel inferior muestra el mensaje del Assert que falló. Por ejemplo: 'Importe negativo debe retornar False — Expected: False, Actual: True'. Ese mensaje describe exactamente qué está mal.
4
Navegar al código desde el Test Explorer
Doble click sobre un test en el Test Explorer → el VBE navega al módulo y función correspondiente. Desde ahí, también podés navegar al módulo bajo test con F12 sobre el nombre de la función.
5
Correr tests de regresión antes de cada entrega
Antes de entregar al cliente: Rubberduck → Test Explorer → Run All. Todos deben estar en verde. Si alguno está en rojo, hay una regresión — algo que funcionaba antes dejó de funcionar.

xlwings permite invocar macros VBA desde Python. Esto hace posible integrar el testing de VBA al pipeline de CI, aunque requiere que la máquina del CI tenga Excel instalado.
1
Instalar xlwings en el entorno de CI
pip install xlwings. El pipeline necesita una máquina Windows con Excel instalado. En Azure DevOps usar un agente self-hosted con Excel, no los agentes Microsoft-hosted (no tienen Excel).
2
Crear el archivo de tests de integración
Crear tests/integration/test_vba_macros.py. Cada test abre el Excel sin interfaz gráfica, ejecuta una macro, y verifica el resultado en el Excel.
3
Separar tests lentos de tests rápidos
Los tests de xlwings son lentos (abren Excel, ejecutan macros). Marcarlos con @pytest.mark.integration para que no corran en cada PR sino solo en el pipeline de noche o de release.
4
Documentar el QA manual en un checklist
Para lo que no se puede automatizar (validación visual, formatos complejos de Excel), crear un documento Word/Excel con el checklist de QA. Cada criterio del PDD tiene una fila con: descripción, cómo verificarlo, y la columna de resultado (OK/FAIL/NA).

"""
Tests de integración VBA via xlwings.
Requiere: Excel instalado, agente CI Windows self-hosted.
Marcar como integration para no correr en cada PR.
"""
import pytest
import xlwings as xw
from pathlib import Path

EXCEL_FILE = Path("data/ReporteTransferPricing.xlsm")


@pytest.fixture(scope="module")
def workbook():
    """Abre el Excel una sola vez para todos los tests del módulo."""
    # visible=False: sin interfaz gráfica (para CI)
    app = xw.App(visible=False, add_book=False)
    wb = app.books.open(str(EXCEL_FILE.resolve()))
    yield wb
    wb.close()
    app.quit()


@pytest.mark.integration
class TestVBAMacros:
    """Tests que invocan macros VBA y verifican el output."""

    def test_macro_generar_reporte_crea_hoja(self, workbook):
        """PASO 1: Ejecutar la macro. PASO 2: Verificar que creó la hoja."""
        # Limpiar estado previo
        try:
            workbook.sheets["Resultado"].delete()
        except Exception:
            pass

        # Ejecutar macro
        workbook.macro("ModuloReporte.GenerarReporte")()

        # Verificar output
        sheet_names = [s.name for s in workbook.sheets]
        assert "Resultado" in sheet_names

    def test_hoja_resultado_tiene_encabezados_del_pdd(self, workbook):
        """Verificar que los encabezados coinciden con los definidos en el PDD."""
        sheet = workbook.sheets["Resultado"]
        # Los encabezados esperados están en el PDD, sección "Estructura del output"
        esperados = ["CUIT", "Razón Social", "Importe Total", "Fecha", "CAE"]
        for col, header in enumerate(esperados, start=1):
            actual = sheet.cells(1, col).value
            assert actual == header, \
                f"Columna {col}: PDD dice '{header}', Excel tiene '{actual}'"

    def test_ningun_importe_es_negativo(self, workbook):
        """Criterio del PDD: todos los importes del output deben ser positivos."""
        sheet = workbook.sheets["Resultado"]
        ultima_fila = sheet.cells(sheet.cells.last_cell.row, 3).end("up").row
        for fila in range(2, ultima_fila + 1):
            importe = sheet.cells(fila, 3).value
            assert importe > 0, f"Fila {fila}: importe negativo o cero: {importe}"

Convención de nombre: MetodoQueTestea_Condición_ResultadoEsperado. Ejemplo: Validar_CuitNulo_LanzaArgumentNullException. Cuando el test falla en el CI, el nombre solo dice exactamente qué está roto.
1
Crear la clase de test con la misma estructura que la clase de producción
Una clase InvoiceValidatorTests para testear InvoiceValidator. Dentro, una región o subclase por cada método público de la clase de producción.
2
Escribir el test [Fact] del happy path
El primer test de cada método es siempre el happy path con [Fact]. Estructura Arrange/Act/Assert con FluentAssertions: resultado.Should().BeTrue("porque...").
3
Agregar tests [Theory] para múltiples inputs
Cuando el mismo comportamiento debe verificarse con varios valores, usar [Theory] con [InlineData]. xUnit corre el método una vez por cada [InlineData].
4
Mockear las dependencias con Moq
Cualquier interfaz que la clase de producción recibe por constructor se mockea: var mock = new Mock<IServicio>(). Configurar con Setup lo que debe devolver. Pasar mock.Object al constructor.
5
Verificar excepciones con FluentAssertions
Para métodos que deben lanzar excepciones: Action accion = () => clase.Metodo(null); accion.Should().Throw<ArgumentNullException>().WithParameterName("invoice").
CSHARP
InvoiceValidatorTests.cs — estructura completa del ciclo
using System;
using Xunit;
using Moq;
using FluentAssertions;
using RpaTeam.Comprobantes;

namespace RpaTeam.Tests.Comprobantes
{
    public class InvoiceValidatorTests
    {
        // ── SHARED SETUP ──────────────────────────────────────
        // Datos de prueba reutilizables en todos los tests de esta clase.
        // No usar [Fact] aquí, solo helpers privados.
        private static Invoice CreateValidInvoice() => new Invoice(
            "20123456789", "0001-00001234", DateTime.Today.AddDays(-1), 12100m);

        private const string ValidCuit = "30987654321";


        // ── HAPPY PATH ────────────────────────────────────────
        [Fact]
        public void Validar_FacturaValida_RetornaTrue()
        {
            // Arrange
            var invoice = CreateValidInvoice();

            // Act
            var resultado = InvoiceValidator.Validar(invoice, ValidCuit);

            // Assert — FluentAssertions: mensaje descriptivo si falla
            resultado.Should().BeTrue("un comprobante con datos válidos debe pasar la validación.");
        }


        // ── VALIDACIÓN DE ARGUMENTOS ──────────────────────────
        [Fact]
        public void Validar_InvoiceNull_LanzaArgumentNullException()
        {
            Action accion = () => InvoiceValidator.Validar(null!, ValidCuit);

            accion.Should().Throw()
                  .WithParameterName("invoice");
        }


        // ── THEORY: múltiples casos con [InlineData] ──────────
        [Theory]
        [InlineData("123")]           // muy corto
        [InlineData("")]              // vacío
        [InlineData("1234567890123")] // muy largo
        public void Validar_CuitReceptorInvalido_LanzaArgumentException(string cuitInvalido)
        {
            var invoice = CreateValidInvoice();

            Action accion = () => InvoiceValidator.Validar(invoice, cuitInvalido);

            accion.Should().Throw()
                  .WithParameterName("cuitReceptor");
        }


        // ── MOCK: aislar dependencia externa ──────────────────
        [Fact]
        public void Descargar_ArcaDevuelveVacio_RetornaListaVacia()
        {
            // Arrange: stub del servicio de ARCA
            var stubArca = new Mock(MockBehavior.Loose);
            stubArca.Setup(s => s.GetInvoices(
                    It.IsAny(), It.IsAny(), It.IsAny()))
                .Returns(new System.Collections.Generic.List());

            var processor = new InvoiceProcessor(stubArca.Object);

            // Act
            var resultado = processor.Descargar("20123456789", DateTime.Today, DateTime.Today);

            // Assert
            resultado.Should().BeEmpty("ARCA devolvió vacío, el resultado debe ser vacío.");
        }
    }
}

Diferencia clave con otros lenguajes: en UiPath el "código" es un diagrama visual de actividades. No se puede testear línea por línea. El foco del QA es verificar que el output del proceso es correcto según los criterios del PDD — no que las actividades internas funcionan de cierta manera.

⚠️
Sin Studio Pro: no hay Test Cases nativos. La estrategia es correr el bot en homologación y usar Robot Framework para verificar el output (archivos Excel, base de datos, emails). El bot se ejecuta como proceso real, RF solo valida lo que produjo.
💡
Los test cases se diseñan a partir de los criterios del PDD, no a partir de los workflows de Studio. La pregunta no es "¿cómo testeo este Invoke Workflow?" sino "¿cómo verifico que el output cumple el criterio X del PDD?"
1
Leer los criterios de aceptación del PDD
Cada criterio del PDD se convierte en al menos un test case. Ejemplos: 'El archivo Excel debe tener las columnas CUIT, Razón Social, Importe, Fecha, CAE' → 1 test case. 'Ningún importe puede ser negativo' → 1 test case.
2
Clasificar los criterios por tipo de test
Smoke: los criterios más críticos (el proceso corre, genera output). Happy path: flujo normal completo. Excepciones: qué pasa con inputs inválidos. Regresión: bugs corregidos que no deben volver.
3
Definir los datos de entrada para cada test
Para cada test case definir: el CUIT, el período, el archivo de entrada, y el output esperado. Documentarlo en el Excel de test data antes de ejecutar. Los datos deben estar en el entorno de homologación.
4
Definir el criterio de éxito campo por campo
No alcanza con 'el proceso terminó sin error'. Para cada test case: qué archivo se genera, qué columnas tiene, cuántas filas, qué valores son aceptables. Esto va en el PDD como 'criterios de aceptación del output'.
5
Crear el Excel de casos de test
Crear un archivo Excel con columnas: ID, Descripción, Datos de entrada, Output esperado, Prioridad (P1/P2/P3), Estado (Pendiente/OK/FAIL). Este archivo es el contrato de QA del sprint.
📘
Con Studio Pro los Test Cases son actividades nativas. Tienen estructura Arrange/Act/Assert dentro de Studio y se integran con Orchestrator para ejecución remota.
1
Crear el proyecto de tests en Studio
File → New → Test Automation Project. Studio crea la estructura con carpeta Tests/ separada de Workflows/. Los Test Cases van en Tests/, los workflows bajo test en Workflows/.
2
Crear un Test Case por criterio del PDD
Tests → New Test Case. Nombrar como: TC-001_DescargaExitosa_PeriodoEnero. Dentro, tres secuencias: Given (preparar datos y mocks), When (invocar el workflow), Then (verificar con Verify Expression).
3
Configurar Mock Activities para dependencias externas
Click derecho sobre la actividad que conecta a ARCA / SAP / portal web → Mock Activity. Configurar el archivo JSON con la respuesta simulada. El test corre sin conectarse al sistema real.
4
Usar Verify Expression para assertions
En la sección Then: agregar actividades Verify Expression. Expresión: out_Comprobantes.Rows.Count > 0. Mensaje: Se esperaban comprobantes en el período.. Si la expresión es False, el test falla con ese mensaje.
5
Correr los tests desde Studio
Test Explorer (View → Test Explorer). Run All corre todos los test cases del proyecto. Los tests verdes pasaron, los rojos fallaron. Click sobre el rojo para ver qué Verify Expression falló.
6
Publicar tests en Orchestrator para CI
Publish el proyecto de tests a Orchestrator como un Process separado. En el pipeline de CI: invocar la API de Orchestrator para triggerear la ejecución y polling hasta que termine.

📘
Sin Studio Pro la estrategia es: (1) correr el bot realmente en homologación, (2) usar Robot Framework para verificar el output que el bot generó. Robot no "entra" al bot — solo verifica los archivos, la base de datos o los emails producidos.
1
Deployar el bot en el entorno de homologación
Publicar el proceso en Orchestrator de homologación. Ejecutar manualmente o via API para el período de prueba. Esperar a que termine. El bot genera su output normalmente.
2
Correr el smoke de Robot Framework
robot --include smoke tests/e2e/robot/suites/. Verifica que el output existe, que el archivo se creó, y que no hay errores CRITICAL en el log del bot. Si el smoke falla, el proceso tuvo un problema crítico.
3
Correr la suite de validación de output
robot tests/e2e/robot/suites/. Los test cases de Robot leen el Excel/DB generado por el bot y verifican cada columna contra los criterios del PDD. Un test por cada criterio.
4
Abrir el log.html y revisar los fallos
Robot genera log.html en la carpeta de resultados. Abrir en el browser. Los tests en rojo muestran exactamente qué columna o valor no cumplió el criterio. Capturar pantalla para el ticket de bug.
5
Registrar el bug con la evidencia del log.html
Por cada test rojo: crear ticket en Azure DevOps con la captura del log.html, el criterio del PDD que se incumplió, y los valores esperados vs. actuales. Asignar prioridad P1/P2/P3.
6
Después del fix: re-correr solo los tests que fallaron
robot --include bug-ID tests/e2e/robot/suites/. Verificar que el fix resolvió el problema. Después, correr la suite completa para verificar que el fix no rompió nada más.
BASH
Terminal — flujo completo sin Studio Pro
# ── DÍA 8: SMOKE Y HAPPY PATH ────────────────────────────────────────────

# 1. El bot ya corrió en homologación (paso manual en Orchestrator)
# Verificar que el proceso terminó: Orchestrator → Jobs → Estado = Successful

# 2. Smoke: verificar que el output existe y es válido a nivel básico
robot --include smoke       --outputdir results/smoke       --variable OUTPUT_DIR:/ruta/al/output/del/bot       tests/e2e/robot/suites/

# Si el smoke pasa → continuar con el happy path
# Si el smoke falla → el bot tuvo un problema grave, revisar Orchestrator logs

# 3. Suite de happy path
robot --include happy-path       --outputdir results/happy-path       tests/e2e/robot/suites/

# Abrir el reporte
start results/happy-path/report.html  # Windows
open results/happy-path/report.html   # Mac

# ── DÍA 9: SUITE COMPLETA ────────────────────────────────────────────────

# 4. Suite completa (happy path + excepciones + regresión)
robot --outputdir results/full       --variable ENTORNO:homologacion       tests/e2e/robot/suites/

# 5. Resumen rápido en terminal
robot --outputdir results/full tests/e2e/robot/suites/
# PASSED  22
# FAILED   1
# → Abrir log.html para ver cuál falló

# 6. Rerun solo los tests que fallaron (más rápido que correr todo de nuevo)
robot --rerunfailed results/full/output.xml       --outputdir results/rerun       tests/e2e/robot/suites/

# 7. Merge de resultados (suite completa + rerun)
rebot --merge results/full/output.xml results/rerun/output.xml       --output results/final/output.xml
