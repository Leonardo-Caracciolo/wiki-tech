Patrón AAA — Arrange · Act · Assert

Todos los tests del equipo siguen el patrón AAA. Es la base de un test legible y mantenible.

// xUnit — patrón AAA con FluentAssertions
using FluentAssertions;
using Moq;
using Xunit;

public class ClienteServiceTests
{
    [Fact]
    public async Task GetClienteAsync_ClienteExiste_RetornaCliente()
    {
        // Arrange
        var expectedCliente = new Cliente { Id = 1, Nombre = "Spotify" };
        var mockRepo = new Mock<IClienteRepository>();
        mockRepo.Setup(r => r.GetByIdAsync(1, It.IsAny<CancellationToken>()))
               .ReturnsAsync(expectedCliente);

        var sut = new ClienteService(mockRepo.Object); // sut = System Under Test

        // Act
        var result = await sut.GetClienteAsync(1);

        // Assert
        result.Should().NotBeNull();
        result!.Id.Should().Be(1);
        result.Nombre.Should().Be("Spotify");
    }

    [Fact]
    public async Task GetClienteAsync_ClienteNoExiste_LanzaNotFoundException()
    {
        // Arrange
        var mockRepo = new Mock<IClienteRepository>();
        mockRepo.Setup(r => r.GetByIdAsync(It.IsAny<int>(), It.IsAny<CancellationToken>()))
               .ReturnsAsync((Cliente?)null);

        var sut = new ClienteService(mockRepo.Object);

        // Act & Assert
        await sut.Invoking(s => s.GetClienteAsync(999))
                .Should().ThrowAsync<NotFoundException>()
                .WithMessage("*999*");
    }
}

Theory — Tests parametrizados

// xUnit — [Theory] con [InlineData]
public class ValidacionTests
{
    [Theory]
    [InlineData("20123456789", true)]
    [InlineData("30987654321", true)]
    [InlineData("123", false)]
    [InlineData("", false)]
    [InlineData(null, false)]
    public void ValidarCuit_VariosInputs_RetornaEsperado(string? cuit, bool expected)
    {
        var result = CuitValidator.Validar(cuit);
        result.Should().Be(expected);
    }

    // Con datos complejos — MemberData
    public static IEnumerable<object[]> DatosClientes => new[]
    {
        new object[] { new Cliente { Activo = true,  Monto = 50_000 }, "B" },
        new object[] { new Cliente { Activo = true,  Monto = 200_000}, "A" },
        new object[] { new Cliente { Activo = false, Monto = 500_000}, "N/A"},
    };

    [Theory]
    [MemberData(nameof(DatosClientes))]
    public void GetSegmento_ClienteVario_RetornaSegmentoEsperado(Cliente c, string expected)
    {
        Segmentador.GetSegmento(c).Should().Be(expected);
    }
}

Moq — Mocking de dependencias

using Moq;

// Setup básico
var mock = new Mock<IEmailService>();

// Retornar valor
mock.Setup(s => s.SendAsync(It.IsAny<string>())).ReturnsAsync(true);

// Lanzar excepción
mock.Setup(s => s.SendAsync("[email protected]"))
    .ThrowsAsync(new InvalidEmailException());

// Verificar llamadas
mock.Verify(s => s.SendAsync(It.IsAny<string>()), Times.Once());
mock.Verify(s => s.SendAsync("[email protected]"), Times.Exactly(2));
mock.VerifyNoOtherCalls(); // ningún otro método fue llamado

// Capturar argumentos con Callback
string? emailCapturado = null;
mock.Setup(s => s.SendAsync(It.IsAny<string>()))
    .Callback<string>(email => emailCapturado = email)
    .ReturnsAsync(true);

// It.Is para condiciones custom
mock.Setup(s => s.GetClienteAsync(It.Is<int>(id => id > 0)))
    .ReturnsAsync(expectedCliente);
